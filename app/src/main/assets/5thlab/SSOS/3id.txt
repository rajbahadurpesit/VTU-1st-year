"int "|"char "|"float "|"double "
	{

		{
			ch=input();
			if(ch==',')
			count++;
			else if(ch==';')
			{
			 count++;
			 break;
			}
		}
	}
.|\n ;

int main(int argc,char *argv[])
{

	{
		printf("Invalid number of arguments!\n");
